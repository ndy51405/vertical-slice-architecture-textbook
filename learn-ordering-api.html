DDD — Value Objects 與 Aggregate

Value Object vs 原始型別
💡 不用 Value Object：Guid customerId 和 Guid orderId 都是 Guid， 編譯器無法阻止你把 customerId 傳到需要 orderId 的地方。

用 Value Object：CustomerId 和 OrderId 是不同的型別， 傳錯型別會在編譯時期就報錯，而不是在執行時期才發現。
簡單 Value Object（ID 類型）
Domain/ValueObjects/OrderId.csC#
// record 天生不可變且以值比較
public record OrderId
{
    public Guid Value { get; }

    private OrderId(Guid value) => Value = value;

    // 工廠方法：集中驗證邏輯
    public static OrderId Of(Guid value)
    {
        ArgumentNullException.ThrowIfNull(value);
        if (value == Guid.Empty)
            throw new DomainException("OrderId cannot be empty");
        return new OrderId(value);
    }
}

// 用法：OrderId.Of(someGuid) 而不是 new OrderId(someGuid)
// 外部無法繞過驗證直接建立

Aggregate Root — Order
Domain/Models/Order.csC#
// Order 是 Aggregate Root，外部只能透過 Order 操作 OrderItem
// 不能直接 new OrderItem() 然後儲存，必須透過 order.Add()
public class Order : Aggregate<OrderId>
{
    private readonly List<OrderItem> _orderItems = [];
    public IReadOnlyList<OrderItem> OrderItems => _orderItems.AsReadOnly();

    public CustomerId CustomerId { get; private set; } = default!;
    public OrderName  OrderName  { get; private set; } = default!;
    public Address    ShippingAddress { get; private set; } = default!;
    public Address    BillingAddress  { get; private set; } = default!;
    public Payment    Payment    { get; private set; } = default!;
    public OrderStatus Status   { get; private set; } = OrderStatus.Draft;
    public decimal TotalPrice => _orderItems.Sum(i => i.Price * i.Quantity);

    // 工廠方法：建立 Order 的唯一合法入口
    public static Order Create(
        OrderId id, CustomerId customerId, OrderName orderName,
        Address shippingAddress, Address billingAddress, Payment payment)
    {
        var order = new Order
        {
            Id = id, CustomerId = customerId, OrderName = orderName,
            ShippingAddress = shippingAddress, BillingAddress = billingAddress,
            Payment = payment
        };

        // 建立後立即發出 Domain Event
        order.AddDomainEvent(new OrderCreatedEvent(order));
        return order;
    }

    // 外部只能透過這個方法新增 OrderItem，確保 Aggregate 的一致性
    public void Add(ProductId productId, int quantity, decimal price)
    {
        var orderItem = new OrderItem(Id, productId, quantity, price);
        _orderItems.Add(orderItem);
    }

    public void Remove(ProductId productId)
    {
        var item = _orderItems
            .FirstOrDefault(i => i.ProductId == productId);
        if (item is not null)
            _orderItems.Remove(item);
    }

    public void Update(
        OrderName orderName, Address shippingAddress,
        Address billingAddress, Payment payment, OrderStatus status)
    {
        OrderName = orderName; ShippingAddress = shippingAddress;
        BillingAddress = billingAddress; Payment = payment; Status = status;
        AddDomainEvent(new OrderUpdatedEvent(this));
    }
}
Aggregate 基底類別
Domain/Abstractions/Aggregate.csC#
public abstract class Aggregate<TId> : Entity<TId>, IAggregate<TId>
{
    private readonly List<IDomainEvent> _domainEvents = [];
    public IReadOnlyList<IDomainEvent> DomainEvents => _domainEvents.AsReadOnly();

    // Order.Create() 和 Order.Update() 呼叫這個方法把事件加入暫存清單
    public void AddDomainEvent(IDomainEvent domainEvent)
        => _domainEvents.Add(domainEvent);

    // DispatchDomainEventsInterceptor 在 SaveChanges 前呼叫這個
    public IDomainEvent[] ClearDomainEvents()
    {
        var events = _domainEvents.ToArray();
        _domainEvents.Clear();
        return events;
    }
}

DispatchDomainEventsInterceptor — 核心機制
💡 這個 Interceptor 讓 Handler 不需要手動發布 Domain Event。 Handler 只需要呼叫 order.Update()，事件會在 SaveChanges 執行前自動被收集並發布。

Infrastructure/Interceptors/DispatchDomainEventsInterceptor.csC#
public class DispatchDomainEventsInterceptor(IMediator mediator)
    : SaveChangesInterceptor
{
    // 攔截 SaveChanges 之前（SavingChanges，不是 SavedChanges）
    // 原因：確保事件和資料在同一個交易範圍內
    public override InterceptionResult<int> SavingChanges(
        DbContextEventData eventData, InterceptionResult<int> result)
    {
        DispatchDomainEvents(eventData.Context).GetAwaiter().GetResult();
        return base.SavingChanges(eventData, result);
    }

    public override async ValueTask<InterceptionResult<int>> SavingChangesAsync(
        DbContextEventData eventData, InterceptionResult<int> result,
        CancellationToken ct = default)
    {
        await DispatchDomainEvents(eventData.Context);
        return await base.SavingChangesAsync(eventData, result, ct);
    }

    private async Task DispatchDomainEvents(DbContext? context)
    {
        if (context is null) return;

        // ① 找出所有 Aggregate（IAggregate 實作者），取出它們的 DomainEvents
        var aggregates = context.ChangeTracker.Entries<IAggregate>()
            .Where(a => a.Entity.DomainEvents.Any())
            .Select(a => a.Entity)
            .ToList();

        var events = aggregates
            .SelectMany(a => a.DomainEvents)
            .ToList();

        // ② 清除 Aggregate 上的事件（避免重複發布）
        aggregates.ForEach(a => a.ClearDomainEvents());

        // ③ 透過 MediatR 發布每個事件
        foreach (var @event in events)
            await mediator.Publish(@event);
    }
}
AuditableEntityInterceptor — 自動填入稽核欄位
Infrastructure/Interceptors/AuditableEntityInterceptor.csC#
public class AuditableEntityInterceptor : SaveChangesInterceptor
{
    public override InterceptionResult<int> SavingChanges(
        DbContextEventData eventData, InterceptionResult<int> result)
    {
        UpdateAuditableEntities(eventData.Context);
        return base.SavingChanges(eventData, result);
    }

    private static void UpdateAuditableEntities(DbContext? context)
    {
        if (context is null) return;

        var entries = context.ChangeTracker.Entries<IEntity>();

        foreach (var entry in entries)
        {
            if (entry.State == EntityState.Added)
            {
                entry.Entity.CreatedBy = "mehmet";  // 正式環境從 ICurrentUser 取得
                entry.Entity.CreatedAt = DateTime.UtcNow;
            }

            if (entry.State == EntityState.Modified ||
                entry.HasChangedOwnedEntities())
            {
                entry.Entity.LastModifiedBy = "mehmet";
                entry.Entity.LastModified = DateTime.UtcNow;
            }
        }
    }
}

Command Handler — 透過 Aggregate 操作
Application/Orders/Commands/CreateOrder/CreateOrderCommandHandler.csC#
public class CreateOrderCommandHandler(IApplicationDbContext dbContext)
    : ICommandHandler<CreateOrderCommand, CreateOrderResult>
{
    public async Task<CreateOrderResult> Handle(
        CreateOrderCommand command, CancellationToken ct)
    {
        // ① 把 DTO 轉成 Value Object（透過工廠方法，包含驗證）
        var order = CreateNewOrder(command.Order);

        // ② 加入 DbContext（EF Core 開始追蹤）
        dbContext.Orders.Add(order);

        // ③ 儲存，Interceptors 在這裡自動執行
        await dbContext.SaveChangesAsync(ct);

        return new CreateOrderResult(order.Id.Value);
    }

    private static Order CreateNewOrder(OrderDto orderDto)
    {
        // 所有 Value Object 都透過 Of() 工廠方法建立，確保驗證
        var shippingAddress = Address.Of(
            orderDto.ShippingAddress.FirstName,
            orderDto.ShippingAddress.LastName,
            orderDto.ShippingAddress.EmailAddress,
            orderDto.ShippingAddress.AddressLine,
            orderDto.ShippingAddress.Country,
            orderDto.ShippingAddress.State,
            orderDto.ShippingAddress.ZipCode);

        var order = Order.Create(
            id:              OrderId.Of(Guid.NewGuid()),
            customerId:      CustomerId.Of(orderDto.CustomerId),
            orderName:       OrderName.Of(orderDto.OrderName),
            shippingAddress: shippingAddress,
            billingAddress:  /* 同上 */ shippingAddress,
            payment:         Payment.Of(...));

        // 透過 Aggregate Root 的方法新增 Items
        foreach (var item in orderDto.OrderItems)
            order.Add(ProductId.Of(item.ProductId), item.Quantity, item.Price);

        return order;
    }
}
Query Handler — 讀取效能優化
Application/Orders/Queries/GetOrders/GetOrdersQueryHandler.csC#
public class GetOrdersQueryHandler(IApplicationDbContext dbContext)
    : IQueryHandler<GetOrdersQuery, GetOrdersResult>
{
    public async Task<GetOrdersResult> Handle(
        GetOrdersQuery query, CancellationToken ct)
    {
        var pageIndex = query.PaginationRequest.PageIndex;
        var pageSize  = query.PaginationRequest.PageSize;

        var total = await dbContext.Orders.LongCountAsync(ct);

        var orders = await dbContext.Orders
            // AsNoTracking：Query 不需要 EF Core 追蹤變更，效能較好
            .AsNoTracking()
            .Include(o => o.OrderItems)
            .OrderBy(o => o.OrderName.Value)
            .Skip(pageIndex * pageSize)
            .Take(pageSize)
            .ToListAsync(ct);

        return new GetOrdersResult(
            new PaginatedResult<OrderDto>(
                pageIndex, pageSize, total,
                orders.ToOrderDtoList()));  // 手動映射
    }
}
為什麼 Query 用手動映射而不是 Mapster
💡 Order 的屬性都是 Value Object（OrderName.Value、CustomerId.Value）， Mapster 自動映射到扁平的 DTO 需要複雜的設定。 手動映射雖然程式碼較多，但更明確、不易出錯，也不依賴 Mapster 的設定。 Command 端的映射（DTO → Value Object）則因為有工廠方法驗證，直接手動建立更安全。

HasConversion — ID Value Object 對應
Infrastructure/Configurations/OrderConfiguration.cs（節錄）C#
public class OrderConfiguration : IEntityTypeConfiguration<Order>
{
    public void Configure(EntityTypeBuilder<Order> builder)
    {
        builder.HasKey(o => o.Id);

        // HasConversion：告訴 EF Core 如何在 C# 型別和資料庫型別之間轉換
        // OrderId（C#）↔ Guid（資料庫）
        // 讀取時：Guid → OrderId.Of(guid)
        // 寫入時：orderId.Value → Guid
        builder.Property(o => o.Id)
            .HasConversion(id => id.Value, value => OrderId.Of(value));

        builder.Property(o => o.CustomerId)
            .HasConversion(id => id.Value, value => CustomerId.Of(value));

        // OrderName：string → OrderName.Value
        // ComplexProperty 展平到同一張資料表，而不是建立子資料表
        builder.ComplexProperty(o => o.OrderName, nameBuilder =>
        {
            nameBuilder.Property(n => n.Value)
                .HasColumnName("OrderName")
                .HasMaxLength(100)
                .IsRequired();
        });

        // Address 展平成多個欄位，兩個 Address 共用相同的結構
        builder.ComplexProperty(o => o.ShippingAddress, addrBuilder =>
        {
            addrBuilder.Property(a => a.FirstName)
                .HasMaxLength(50).IsRequired();
            addrBuilder.Property(a => a.ZipCode)
                .HasMaxLength(5);
            // ... 其他欄位
        });

        builder.ComplexProperty(o => o.BillingAddress, addrBuilder =>
        {
            // 和 ShippingAddress 相同的結構，但欄位名稱不同
        });

        // OrderStatus 以字串儲存（"Draft"、"Pending"）而非數字
        // 好處：資料庫可讀性高，新增 enum 值不影響現有資料
        builder.Property(o => o.Status)
            .HasDefaultValue(OrderStatus.Draft)
            .HasConversion(s => s.ToString(), value =>
                (OrderStatus)Enum.Parse(typeof(OrderStatus), value));

        // OrderItem 一對多關係
        builder.HasMany(o => o.OrderItems)
            .WithOne()
            .HasForeignKey("OrderId");  // Shadow property
    }
}
Interceptor 注入到 DbContext
Infrastructure/DependencyInjection.csC#
public static IServiceCollection AddInfrastructureServices(
    this IServiceCollection services,
    IConfiguration configuration)
{
    // ① 先把 Interceptors 獨立註冊為 Singleton
    services.AddSingleton<AuditableEntityInterceptor>();
    services.AddSingleton<DispatchDomainEventsInterceptor>();

    services.AddDbContext<ApplicationDbContext>((sp, options) =>
    {
        // ② 透過 Service Provider 取得 Interceptors
        var auditInterceptor   = sp.GetRequiredService<AuditableEntityInterceptor>();
        var dispatchInterceptor = sp.GetRequiredService<DispatchDomainEventsInterceptor>();

        options
            .AddInterceptors(auditInterceptor, dispatchInterceptor)
            .UseSqlServer(configuration.GetConnectionString("Database"));
    });

    // ③ 讓 Application 層透過介面存取 DbContext
    services.AddScoped<IApplicationDbContext,
        ApplicationDbContext>();

    return services;
}

// IConsumer<T> 是 MassTransit 的介面，不是 MediatR
// 當 RabbitMQ 收到 BasketCheckoutEvent，MassTransit 自動呼叫 Consume()
public class BasketCheckoutEventHandler(ISender sender)
    : IConsumer<BasketCheckoutEvent>
{
    public async Task Consume(ConsumeContext<BasketCheckoutEvent> context)
    {
        // 把 Integration Event 轉成 Command，交給 MediatR 處理
        // 這樣 Consumer 不需要重複實作建立訂單的邏輯
        // 和 POST /orders 走同一條 Handler 路徑
        var command = MapToCreateOrderCommand(context.Message);
        await sender.Send(command);
    }

    private static CreateOrderCommand MapToCreateOrderCommand(BasketCheckoutEvent message)
    {
        var addressDto = new AddressDto(
            message.FirstName, message.LastName, message.EmailAddress,
            message.AddressLine, message.Country, message.State, message.ZipCode);

        var paymentDto = new PaymentDto(
            message.CardName, message.CardNumber,
            message.Expiration, message.CVV, message.PaymentMethod);

        // 注意：OrderItems 是硬編碼的商品 ID
        // 因為 BasketCheckoutEvent 裡只有商品名稱和價格，沒有商品 ID
        // 這是這個教學專案的簡化設計
        var orderId = Guid.NewGuid();
        return new CreateOrderCommand(new OrderDto(
            orderId,
            message.CustomerId,
            message.UserName,
            addressDto,
            addressDto,
            paymentDto,
            OrderStatus.Pending,
            new List<OrderItemDto>
            {
                new(Guid.Parse("5334c996-8457-4cf0-815c-ed2b77c4ff61"), 2, 500), // iPhone X
                new(Guid.Parse("c67d6323-e8b1-4bdf-9a75-b0d0d2e7e914"), 1, 500)  // Samsung 10
            }
        ));
    }
}
Feature Management — OrderCreatedEventHandler

public class OrderCreatedEventHandler(
    IPublishEndpoint publishEndpoint,
    IFeatureManager featureManager,
    ILogger<OrderCreatedEventHandler> logger)
    : INotificationHandler<OrderCreatedEvent>
{
    public async Task Handle(OrderCreatedEvent notification, CancellationToken ct)
    {
        logger.LogInformation("OrderCreatedEvent handled: {OrderId}",
            notification.Order.Id);

        // Feature Flag 控制：只有在 "OrderFullfilment" = true 時才發布
        // 現在設定為 false，所以不會觸發後續的出貨流程
        // 未來啟用只需要改設定檔，不需要改程式碼
        if (await featureManager.IsEnabledAsync("OrderFullfilment"))
        {
            var orderDto = notification.Order.ToOrderDto();
            await publishEndpoint.Publish(orderDto, ct);
        }
    }
}
